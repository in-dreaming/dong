CSS tab-size Property Test

Test 1: Different tab-size Values

The same code with different tab-size settings:

tab-size: 2
function hello() {
	if (true) {
		console.log("Tab size: 2");
	}
}
tab-size: 4
function hello() {
	if (true) {
		console.log("Tab size: 4");
	}
}
tab-size: 8 (default)
function hello() {
	if (true) {
		console.log("Tab size: 8");
	}
}
tab-size: 16
function hello() {
	if (true) {
		console.log("Tab size: 16");
	}
}
